33/// Loads Fluent localization files from a specified directory and creates a
34/// lazily-initialized static instance of `i18n::Locales`.
35///
36/// This macro reads `.ftl` files from subdirectories of the given path (each
37/// subdirectory representing a locale). It parses these files at compile time
38/// and embeds their content into your binary.
39///
40/// The generated static instance provides methods to query for localized messages.
41///
42/// # Syntax
43///
44/// `load!(path: LitStr [, fallback_lang: LitStr] [, check_keys: bool] [, name: Ident] [, on_error: Expr])`
45///
46/// # Arguments
47///
48/// - `path`: A string literal representing the path to the locales directory.
49///   This path should be relative to your crate root (where `Cargo.toml` is).
50///
51/// - `fallback_lang` (optional): A string literal representing the language identifier
52///   (e.g., "en-US") to use as a fallback if a message is not found in the
53///   requested language. Defaults to `"en-US"`.
54///
55/// - `check_keys` (optional): A boolean literal (`true` or `false`). If `true`
56///   (default), the macro will perform a compile-time check to ensure all
57///   locale files have a consistent set of message keys. If `false`, this
58///   check is skipped.
59///
60/// - `name` (optional): An identifier to use as the name for the generated
61///   `lazy_static` variable. Defaults to `LOCALES`.
62///
63/// - `on_error` (optional): An expression that evaluates to a function or closure
64///   to be called when an error occurs during localization (e.g., missing message).
65///   The function should have the signature `fn(errors: &[i18n::FluentError])`.
66///
67/// # Usage
68///
69/// ```ignore
70/// // Basic usage with default values.
71/// i18n::load!("../tests/i18n", name = LOCALES_DEFAULT);
72///
73/// // With a custom fallback language, disabled key checking, and a custom name.
74/// i18n::load!(
75///     "../tests/i18n",
76///     fallback_lang = "hr-hr",
77///     check_keys = false,
78///     name = MY_I18N_DATA
79/// );
80///
81/// // With an error handler.
82/// fn on_error(errors: &[i18n::FluentError]) {
83///     // Log the error, send it to a monitoring service, etc.
84///     println!("Localization errors: {:?}", errors);
85/// }
86///
87/// i18n::load!("../tests/i18n", on_error = on_error, name = LOCALES_WITH_ERROR_HANDLER);
88///
89/// // Example of accessing the generated data (assuming default name `LOCALES_DEFAULT`).
90/// let lang_en = i18n::langid!("en-US");
91/// let query = i18n::Query::new("foo-a");
92/// let message = LOCALES_DEFAULT.query(&lang_en, &query).unwrap();
93/// assert_eq!(message.value, "English A".to_string());
94///
95/// let lang_hr = i18n::langid!("hr-hr");
96/// let query = i18n::Query::new("foo-a");
97/// let message = LOCALES_DEFAULT.query(&lang_hr, &query).unwrap();
98/// assert_eq!(message.value, "Croatian A".to_string());
99/// ```
100#[proc_macro]
101pub fn load(input: TokenStream) -> TokenStream {
102    load::load_impl(input)
103}

105/// Queries for a localized message, returning an `i18n::Message` instance.
106///
107/// This macro provides a convenient way to query for a message using a language
108/// identifier, message ID, and optional arguments. It handles error cases by
109/// providing a default fallback message.
110///
111/// # Syntax
112///
113/// `tr!(lang: Expr, id: LitStr [, locales = VAR_NAME] [, key = value]* [, .attribute_name(key = value)* ])`
114///
115/// - `lang`: A Rust expression that evaluates to a `&LanguageIdentifier` (e.g., `langid!("en-US")` or a variable). This is the language to query for.
116/// - `id`: A string literal representing the ID of the Fluent message.
117/// - `locales` (optional): An identifier for the `i18n::Locales` static variable to use. Defaults to `LOCALES`.
118/// - `key = value`: Optional key-value pairs for arguments to the main message.
119///   `key` must be an string literal, and `value` can be any Rust expression.
120/// - `attr(attr_id, key = value)`: Optional arguments for a specific attribute
121///   of the message. `attr_id` is a string literal representing the attribute ID (e.g., "aria-label" or "attr-arg").
122///
123/// # Returns
124///
125/// An `i18n::Message` instance.
126///
127/// # Examples
128///
129/// ```ignore
130/// // Load the localization data.
131/// i18n::load!("../tests/i18n", fallback_lang = "en-US", name = TR_LOCALES);
132///
133/// let lang_en = i18n::langid!("en-US");
134///
135/// // Basic usage:
136/// let msg1 = i18n::tr!(lang_en, "foo-a", locales = TR_LOCALES);
137/// assert_eq!(msg1.value, "English A".to_string());
138///
139/// // With main message arguments:
140/// let msg2 = i18n::tr!(lang_en, "welcome-back", "username" = "Alice", locales = TR_LOCALES);
141/// assert_eq!(msg2.value, "Welcome back, \u{2068}Alice\u{2069}!".to_string());
142///
143/// // With attribute arguments:
144/// let msg3 = i18n::tr!(lang_en, "login-btn", attr("attr-arg", "text" = "some text"), locales = TR_LOCALES);
145/// assert_eq!(msg3.attrs.get("attr-arg"), Some(&"This is an attribute argument with arbitrary text: \u{2068}some text\u{2069}".to_string()));
146///
147/// // With a custom locales variable:
148/// i18n::load!("../tests/i18n", fallback_lang = "en-US", name = MY_APP_LOCALES);
149/// let msg4 = i18n::tr!(lang_en, "foo-b", locales = MY_APP_LOCALES);
150/// assert_eq!(msg4.value, "English B".to_string());
151/// ```
152#[proc_macro]
153pub fn tr(input: TokenStream) -> TokenStream {
154    tr::tr_impl(input)
155}

157/// Queries an attribute from an existing `i18n::Message` instance.
158///
159/// This macro provides a convenient way to lazily query for an attribute's
160/// value from a `Message` struct that you already have. If the query fails,
161/// it will invoke the `on_error` handler on the `Locales` instance.
162///
163/// # Syntax
164///
165/// `attr!(message_expr, attr_id: LitStr [, locales = VAR_NAME] [, key = value]*)`
166///
167/// - `message_expr`: An expression that evaluates to a `&mut i18n::Message`.
168/// - `attr_id`: A string literal representing the ID of the attribute to query.
169/// - `locales` (optional): The `i18n::Locales` instance to use for the error callback. Defaults to `LOCALES`.
170/// - `key = value`: Optional key-value pairs for arguments to the attribute.
171///
172/// # Returns
173///
174/// A `String`. If the attribute doesn't exist or an error occurs during formatting,
175/// it returns the attribute ID as a fallback string.
176///
177/// # Example
178///
179/// ```ignore
180/// i18n::load!("../tests/i18n", name = ATTR_MACRO_LOCALES);
181/// let lang = i18n::langid!("en-US");
182///
183/// let mut msg = i18n::tr!(lang, "login-btn", locales = ATTR_MACRO_LOCALES);
184///
185/// // Get an attribute with no arguments (using default LOCALES)
186/// let idle_text = i18n::attr!(msg, "idle");
187/// assert_eq!(idle_text, "Login");
188///
189/// // Get an attribute with arguments and a custom locales instance
190/// let arg_text = i18n::attr!(msg, "attr-arg", locales = ATTR_MACRO_LOCALES, "text" = "some value");
191/// assert_eq!(arg_text, "This is an attribute argument with arbitrary text: \u{2068}some value\u{2069}");
192/// ```
193#[proc_macro]
194pub fn attr(input: TokenStream) -> TokenStream {
195    attr::attr_impl(input)
