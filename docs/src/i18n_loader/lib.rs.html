10/// A thread-safe container for all loaded localization data.
11///
12/// It manages multiple `Locale` instances, keyed by language identifier,
13/// and provides a unified interface for querying translations. It also handles
14/// fallback logic to a default language if a translation is missing.
15pub struct Locales {
16    /// The map from a language identifier to its `Locale`.
17    locales: HashMap<LanguageIdentifier, Locale>,
18    /// The language to use as a fallback if a message is not found in the requested language.
19    fallback_lang: LanguageIdentifier,
20    /// An optional error handler to be called with any localization errors.
21    on_error: Option<fn(&[FluentError])>,
22}
23
24impl Locales {
25    /// Creates a new, empty `Locales` collection.
26    ///
27    /// # Arguments
28    /// * `fallback_lang`: The language identifier to use if a translation is not found in the current language.
29    /// * `on_error`: An optional callback function that will be invoked with any errors that occur during message formatting.
30    pub fn new(fallback_lang: LanguageIdentifier, on_error: Option<fn(&[FluentError])>) -> Self {
31        Self {
32            locales: Default::default(),
33            fallback_lang,
34            on_error,
35        }
36    }
37
38    /// Creates a new `Locales` collection from a network resource.
39    ///
40    /// # Arguments
41    /// * `url`: The URL from which to fetch the translation.
42    /// * `fallback_lang`: The language identifier to use if a translation is not found in the current language.
43    /// * `on_error`: An optional callback function that will be invoked with any errors that occur during message formatting.
44    #[cfg(feature = "net")]
45    pub async fn from_url(
46        url: &str,
47        fallback_lang: LanguageIdentifier,
48        on_error: Option<fn(&[FluentError])>,
49    ) -> Result<Self, NetError> {
50        let https = hyper_tls::HttpsConnector::new();
51        let client = hyper::Client::builder().build::<_, hyper::Body>(https);
52
53        let uri = url.parse().unwrap();
54
55        let res = client
56            .get(uri)
57            .await
58            .map_err(NetError::ServerError)?;
59        let body = hyper::body::to_bytes(res.into_body())
60            .await
61            .map_err(NetError::ServerError)?;
62        let definitions: HashMap<String, String> =
63            serde_json::from_slice(&body).map_err(NetError::InvalidFormat)?;
64        let mut parser_errors: Vec<ParserError> = Vec::default();
65        let mut locales: HashMap<LanguageIdentifier, Locale> = HashMap::default();
66        for (langid, definition) in definitions.into_iter() {
67            let langid = match langid.parse::<LanguageIdentifier>() {
68                Ok(langid) => langid,
69                Err(_) => {
70                    parser_errors.push(ParserError::InvalidLangid { langid });
71                    continue;
72                }
73            };
74            let resource = match FluentResource::try_new(definition) {
75                Ok(resource) => resource,
76                Err((_, errors)) => {
77                    parser_errors.push(ParserError::ParserError { langid, errors });
78                    continue;
79                }
80            };
81            locales.insert(langid.clone(), Locale::new(langid, vec![resource]));
82        }
83
84        if !parser_errors.is_empty() {
85            return Err(NetError::ParserError(parser_errors));
86        }
87
88        Ok(Self {
89            locales,
90            fallback_lang,
91            on_error,
92        })
93    }
94
95    /// Adds a new language's localization data to the collection.
96    ///
97    /// # Arguments
98    /// * `lang_str`: A string slice representing the language identifier (e.g., "en-US", "de").
99    /// * `resources`: A vector of `FluentResource`s containing the translation data for this language.
100    ///
101    /// # Panics
102    /// Panics if `lang_str` is not a valid language identifier.
103    pub fn add_locale(&mut self, lang_str: &str, resources: Vec<FluentResource>) {
104        let lang_id: LanguageIdentifier = lang_str.parse().expect("Language ID should be valid");
105        let locale = Locale::new(lang_id.clone(), resources);
106        self.locales.insert(lang_id, locale);
107    }
108
109    /// Queries for a message in a specific language, applying fallback logic if the language is not found.
110    ///
111    /// It first attempts to find the `Locale` for the requested language. If the entire `Locale` is missing,
112    /// it will automatically retry the query using the configured fallback language.
113    #[track_caller]
114    pub fn query(
115        &self,
116        lang: &LanguageIdentifier,
117        query: &Query,
118    ) -> Result<Message, Vec<FluentError>> {
119        let query_result = match self.locales.get(lang) {
120            Some(locale) => locale.query(query),
121            None => {
122                let fallback_locale = self
123                    .locales
124                    .get(&self.fallback_lang)
125                    .expect("a fallback language should *always* exist and be present as a locale");
126                fallback_locale.query(query)
127            }
128        };
129
130        // inspect the errors if on_error exists
131        if let (Some(on_error), Err(errs)) = (&self.on_error, &query_result) {
132            on_error(errs);
133        }
134        query_result
135    }
136
137    /// If an `on_error` handler is configured, this method invokes it with the provided slice of `FluentError`s.
138    pub fn call_on_error(&self, errors: &[FluentError]) {
139        if let Some(on_error) = self.on_error {
140            on_error(errors);
141        }
142    }
143}

145/// Manages Fluent localization resources for a specific locale.
146///
147/// A `Locale` holds a collection of `FluentResource` objects, which contain the
148/// parsed data from `.ftl` (Fluent Translation List) files. It is responsible
149/// for a single language and provides the resources needed to format localized
150/// messages.
151pub struct Locale {
152    /// The underlying `FluentBundle` that manages the collection of resources
153    /// and handles the formatting of messages.
154    bundle: Arc<FluentBundle<Arc<FluentResource>>>,
155}
156
157impl Locale {
158    /// Creates a new `Locale` for a given language and its resources.
159    ///
160    /// # Arguments
161    /// * `lang`: The `LanguageIdentifier` for this locale.
162    /// * `resources`: A vector of `FluentResource`s containing the translation data.
163    pub fn new(lang: LanguageIdentifier, resources: Vec<FluentResource>) -> Self {
164        let mut bundle = FluentBundle::new_concurrent(vec![lang.clone()]);
165        for resource in resources.into_iter() {
166            bundle
167                .add_resource(Arc::new(resource))
168                .expect("resource should never be overriding another; consider this a bug if it happens and open an issue at https://github.com/Instant-Reactive-Systems/i18n/issues");
169        }
170        let bundle = Arc::new(bundle);
171
172        Self { bundle }
173    }
174
175    /// Resolves a `Query` into a fully formatted `Message`.
176    ///
177    /// This method takes a `Query` which specifies a message ID and any
178    /// arguments, and attempts to format it into a `Message` struct.
179    /// If the message ID is not found, or if any errors occur during formatting,
180    /// an `Err` containing a vector of `FluentError`s is returned.
181    #[track_caller]
182    pub fn query(&self, query: &Query) -> Result<Message, Vec<FluentError>> {
183        let mut errors = Vec::default();
184        let msg = match self.bundle.get_message(&query.id) {
185            Some(msg) => msg,
186            None => {
187                errors.push(FluentError::ResolverError(ResolverError::Reference(
188                    ReferenceKind::Message {
189                        id: query.id.to_string(),
190                        attribute: None,
191                    },
192                )));
193                return Err(errors);
194            }
195        };
196
197        let value = match msg.value() {
198            Some(pattern) => self
199                .bundle
200                .format_pattern(pattern, Some(&query.args), &mut errors)
201                .to_string(),
202            None => format!("<{}>", query.id),
203        };
204
205        let mut attrs = HashMap::default();
206        for attr in msg.attributes() {
207            let mut local_errors = Vec::default();
208            let pattern = attr.value();
209            let attr_cache = match query.attr_args.get(attr.id()) {
210                Some(args) => {
211                    let value = self
212                        .bundle
213                        .format_pattern(pattern, Some(args), &mut local_errors);
214
215                    AttrCache {
216                        entry_id: query.id.to_string(),
217                        attr_id: attr.id().to_string(),
218                        value: Some(value.to_string()),
219                        bundle: self.bundle.clone(),
220                    }
221                }
222                None => {
223                    let mut even_more_local_errors = Vec::default();
224                    let value =
225                        self.bundle
226                            .format_pattern(pattern, None, &mut even_more_local_errors);
227
228                    let value = if !even_more_local_errors.is_empty() {
229                        let only_missing_attr_args = even_more_local_errors.iter().all(|err| {
230                            matches!(
231                                err,
232                                FluentError::ResolverError(ResolverError::Reference(
233                                    ReferenceKind::Variable { .. }
234                                ))
235                            )
236                        });
237
238                        // only consider errors other than a missing placeable as an actual error
239                        if !only_missing_attr_args {
240                            local_errors.extend(even_more_local_errors.into_iter());
241                        }
242
243                        None
244                    } else {
245                        Some(value.to_string())
246                    };
247
248                    AttrCache {
249                        entry_id: query.id.to_string(),
250                        attr_id: attr.id().to_string(),
251                        value,
252                        bundle: self.bundle.clone(),
253                    }
254                }
255            };
256
257            attrs.insert(attr.id().to_string(), attr_cache);
258        }
259
260        if !errors.is_empty() {
261            return Err(errors);
262        }
263
264        Ok(Message {
265            id: query.id.to_string(),
266            value: value.to_string(),
267            attrs,
268        })
269    }
270}

272/// Represents a localized message with its ID, value, and attributes.
273#[derive(Debug, PartialEq, Default)]
274pub struct Message {
275    /// The unique identifier for the message (e.g., "login-button").
276    pub id: String,
277    /// The translated main text of the message.
278    pub value: String,
279    /// A map of associated attributes for the message, such as `aria-label`.
280    pub attrs: HashMap<String, AttrCache>,
281}

283/// Represents a request to format a localized message, including its ID and arguments.
284#[derive(Debug, Default)]
285pub struct Query<'a> {
286    /// The ID of the message to format (e.g., "hello-world").
287    id: Cow<'a, str>,
288    /// Arguments for the main message value.
289    args: FluentArgs<'a>,
290    /// Arguments for specific message attributes, keyed by attribute name.
291    attr_args: HashMap<Cow<'a, str>, FluentArgs<'a>>,
292    /// A flag to indicate whether to use the fallback language if the message is not found.
293    with_fallback: bool,
294}
295
296impl<'a> Query<'a> {
297    /// Creates a new `Query` for a given message ID.
298    pub fn new(id: impl Into<Cow<'a, str>>) -> Self {
299        Self {
300            id: id.into(),
301            args: Default::default(),
302            attr_args: Default::default(),
303            with_fallback: false,
304        }
305    }
306
307    /// Adds an argument for the main message value.
308    ///
309    /// # Example
310    ///
311    /// ```
312    /// use i18n_loader::Query;
313    ///
314    /// let query = Query::new("hello-user").with_arg("userName", "Alex");
315    /// ```
316    pub fn with_arg<I, V>(mut self, id: I, value: V) -> Self
317    where
318        I: Into<Cow<'a, str>>,
319        V: Into<FluentValue<'a>>,
320    {
321        self.args.set(id.into(), value.into());
322        self
323    }
324
325    /// Adds an argument for a specific attribute of the message.
326    ///
327    /// # Example
328    ///
329    /// ```
330    /// use i18n_loader::Query;
331    ///
332    /// let query = Query::new("user-tooltip").with_attr_arg("aria-label", "userName", "Alex");
333    /// ```
334    pub fn with_attr_arg<A, I, V>(mut self, attr: A, id: I, value: V) -> Self
335    where
336        A: Into<Cow<'a, str>>,
337        I: Into<Cow<'a, str>>,
338        V: Into<FluentValue<'a>>,
339    {
340        self.attr_args
341            .entry(attr.into())
342            .or_default()
343            .set(id.into(), value.into());
344        self
345    }
346
347    /// Enables or disables fallback to the default language for this specific query.
348    ///
349    /// If set to `true`, and the requested message is not found in the primary language,
350    /// the query will be re-attempted using the `Locales` fallback language.
351    pub fn with_fallback(mut self, enable_fallback: bool) -> Self {
352        self.with_fallback = enable_fallback;
353        self
354    }
355}

357/// Provides a cache for the value of a specific attribute from a localization entry.
358///
359/// This struct is designed for lazy evaluation. It stores the identifiers for a
360/// message's attribute (`entry_id`, `attr_id`) and a static reference to the `Locales`
361/// instance. The actual localized string is only fetched from the origin `Locales`
362/// and stored in `value` upon first access, reducing overhead for attributes that
363/// are not immediately needed.
364pub struct AttrCache {
365    /// The ID of the main localization entry.
366    pub entry_id: String,
367    /// The ID of the attribute entry.
368    pub attr_id: String,
369    /// The cached value of the localization.
370    pub value: Option<String>,
371    /// The underlying `FluentBundle` that manages the collection of resources
372    /// and handles the formatting of messages.
373    pub bundle: Arc<FluentBundle<Arc<FluentResource>>>,
374}
375
376impl AttrCache {
377    /// Queries the cached attribute, formatting it with the given arguments.
378    ///
379    /// If the attribute value is already cached, it will be returned immediately.
380    /// Otherwise, it will be formatted using the provided `args`. If `args` are
381    /// not provided, any arguments required by the attribute will be missing,
382    /// potentially resulting in a formatting error.
383    ///
384    /// # Errors
385    /// Returns a `Vec<FluentError>` if any errors occur during formatting, such as
386    /// missing message IDs, attributes, or arguments.
387    pub fn query(&mut self, args: Option<&FluentArgs>) -> Result<String, Vec<FluentError>> {
388        // return the cached localization
389        if let Some(value) = self.value.clone() {
390            return Ok(value);
391        }
392
393        let mut errors = Vec::default();
394        let msg = match self.bundle.get_message(&self.entry_id) {
395            Some(msg) => msg,
396            None => {
397                errors.push(FluentError::ResolverError(ResolverError::Reference(
398                    ReferenceKind::Message {
399                        id: self.entry_id.to_string(),
400                        attribute: None,
401                    },
402                )));
403                return Err(errors);
404            }
405        };
406
407        let Some(this_attr) = msg.attributes().find(|attr| attr.id() == self.attr_id) else {
408            errors.push(FluentError::ResolverError(ResolverError::Reference(
409                ReferenceKind::Message {
410                    id: self.entry_id.to_string(),
411                    attribute: Some(self.attr_id.to_string()),
412                },
413            )));
414            return Err(errors);
415        };
416
417        let pattern = this_attr.value();
418        let value = self.bundle.format_pattern(pattern, args, &mut errors);
419
420        if !errors.is_empty() {
421            return Err(errors);
422        }
423
424        Ok(value.to_string())
425    }
426}

428/// Implements equality for `AttrCache`.
429///
430/// Two `AttrCache` instances are considered equal if they point to the same
431/// message entry and attribute. The cached `value` is not considered in the
432/// comparison.
433impl PartialEq for AttrCache {
434    fn eq(&self, other: &Self) -> bool {
435        self.entry_id == other.entry_id && self.attr_id == other.attr_id
436    }
437}

439/// Custom debug implementation for `AttrCache`.
440///
441/// This implementation provides a clean debug output, showing the entry ID,
442/// attribute ID, and the cached value if present.
443impl std::fmt::Debug for AttrCache {
444    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
445        f.debug_struct("AttrCache")
446            .field("entry_id", &self.entry_id)
447            .field("attr_id", &self.attr_id)
448            .field("value", &self.value)
449            .finish()
450    }
451}

466        langid: LanguageIdentifier,
